###FOR LAGENARIA HETEROGRAFT MOBILE GENE ANALYSIS
cd ~/Karthik
mkdir -p blast_out fasta bed genes counts results

makeblastdb \
-in USVL1VR-Ls_genome_v1.fa \
-dbtype nucl \
-out LG_db

ls -lh LG_db.*

for s in L1 L2 L3 S1 S2 S3
do
  seqtk seq -a unmapped_fastq/${s}.unmapped_R1.fq.gz > fasta/${s}_R1.fa
  seqtk seq -a unmapped_fastq/${s}.unmapped_R2.fq.gz > fasta/${s}_R2.fa
done


for s in L1 L2 L3 S1 S2 S3
do
  blastn \
  -query fasta/${s}_R1.fa \
  -db LG_db \
  -out blast_out/${s}_R1_vs_LG.tsv \
  -outfmt "6 qseqid sseqid pident length mismatch gapopen qstart qend sstart send evalue bitscore" \
  -evalue 1e-10 \
  -num_threads 12

  blastn \
  -query fasta/${s}_R2.fa \
  -db LG_db \
  -out blast_out/${s}_R2_vs_LG.tsv \
  -outfmt "6 qseqid sseqid pident length mismatch gapopen qstart qend sstart send evalue bitscore" \
  -evalue 1e-10 \
  -num_threads 12
done

wc -l blast_out/*_vs_LG.tsv

for s in L1 L2 L3 S1 S2 S3
do
  awk '$3>=95 && $4>=50' blast_out/${s}_R1_vs_LG.tsv > blast_out/${s}_R1.filtered.tsv
  awk '$3>=95 && $4>=50' blast_out/${s}_R2_vs_LG.tsv > blast_out/${s}_R2.filtered.tsv
done

wc -l blast_out/*.filtered.tsv

for s in L1 L2 L3 S1 S2 S3
do
  echo "${s} R1 unique reads:"
  cut -f1 blast_out/${s}_R1.filtered.tsv | sort -u | wc -l

  echo "${s} R2 unique reads:"
  cut -f1 blast_out/${s}_R2.filtered.tsv | sort -u | wc -l
done


for s in L1 L2 L3 S1 S2 S3
do
  sort -k1,1 -k12,12gr blast_out/${s}_R1.filtered.tsv | sort -u -k1,1 > blast_out/${s}_R1.besthit.tsv
  sort -k1,1 -k12,12gr blast_out/${s}_R2.filtered.tsv | sort -u -k1,1 > blast_out/${s}_R2.besthit.tsv
done


wc -l blast_out/*.besthit.tsv

for s in L1 L2 L3 S1 S2 S3
do
  awk 'BEGIN{OFS="\t"}{
    chr=$2; s=$9; e=$10;
    if(s>e){t=s;s=e;e=t}
    print chr, s-1, e, $1
  }' blast_out/${s}_R1.besthit.tsv > bed/${s}_R1.bed

  awk 'BEGIN{OFS="\t"}{
    chr=$2; s=$9; e=$10;
    if(s>e){t=s;s=e;e=t}
    print chr, s-1, e, $1
  }' blast_out/${s}_R2.besthit.tsv > bed/${s}_R2.bed
done

awk '$3=="gene"' USVL1VR-Ls_v1.gff3 > genes/LG.genes_only.gff3

ls bed | head

ls bed/*.bed | wc -l

 for s in L1 L2 L3 S1 S2 S3
do
  bedtools intersect -a bed/${s}_R1.bed -b genes/LG.genes_only.gff3 -wa -wb \
  | awk 'BEGIN{FS=OFS="\t"}{
      attr=$13
      gsub(/.*ID=/,"",attr)
      gsub(/;.*/,"",attr)
      print $4, attr
    }' \
  | awk '!seen[$1]++' \
  > genes/${s}_R1.read_gene.tsv

  bedtools intersect -a bed/${s}_R2.bed -b genes/LG.genes_only.gff3 -wa -wb \
  | awk 'BEGIN{FS=OFS="\t"}{
      attr=$13
      gsub(/.*ID=/,"",attr)
      gsub(/;.*/,"",attr)
      print $4, attr
    }' \
  | awk '!seen[$1]++' \
  > genes/${s}_R2.read_gene.tsv
done

for f in genes/*_R1.read_gene.tsv genes/*_R2.read_gene.tsv
do
  dups=$(cut -f1 "$f" | sort | uniq -d | wc -l)
  echo "$f : $dups duplicated read IDs"
done

for s in L1 L2 L3 S1 S2 S3
do
  cat genes/${s}_R1.read_gene.tsv genes/${s}_R2.read_gene.tsv \
  | cut -f2 \
  | sort -u \
  > genes/${s}.genes
done


wc -l genes/*.genes

for s in L1 L2 L3 S1 S2 S3
do
  cat genes/${s}_R1.read_gene.tsv genes/${s}_R2.read_gene.tsv \
  | cut -f2 \
  | sort \
  | uniq -c \
  | awk 'BEGIN{OFS="\t"}{print $2,$1}' \
  > counts/${s}.gene_readcounts.tsv
done

ls counts/*.gene_readcounts.tsv | wc -l

head counts/L1.gene_readcounts.tsv

awk '$2 !~ /^[0-9]+$/ {print}' counts/L1.gene_readcounts.tsv | head

for f in counts/*.gene_readcounts.tsv
do
  total=$(awk '{sum+=$2} END{print sum}' "$f")
  echo "$f total_counts: $total"
done

cat genes/L1.genes genes/L2.genes genes/L3.genes | sort -u > results/L_all.genes
cat genes/S1.genes genes/S2.genes genes/S3.genes | sort -u > results/S_all.genes

cat genes/L1.genes genes/L2.genes genes/L3.genes \
| sort \
| uniq -c \
| awk '$1>=2 {print $2}' \
> results/L_rep2.genes

cat genes/S1.genes genes/S2.genes genes/S3.genes | sort -u > results/S_background.genes

comm -23 <(sort results/L_rep2.genes) <(sort results/S_background.genes) > results/L_mobile_final.genes

wc -l results/L_rep2.genes results/S_background.genes results/L_mobile_final.genes

wc -l results/*.genes

run_deseq2 <- function(matrix_file, het_samples, self_samples, output_prefix) {
  
  cat("\n========================================\n")
  cat("Running DESeq2 for:", output_prefix, "\n")
  cat("========================================\n")
  
  count_matrix <- read.table(matrix_file, header = TRUE,
                             row.names = 1, sep = "\t")
  count_matrix <- as.matrix(count_matrix)
  mode(count_matrix) <- "integer"
  cat("Matrix dimensions:", nrow(count_matrix), "genes x", ncol(count_matrix), "samples\n")
  
  colData <- data.frame(
    condition = factor(
      c(rep("heterograft", length(het_samples)),
        rep("selfgraft",   length(self_samples))),
      levels = c("selfgraft", "heterograft")
    ),
    row.names = c(het_samples, self_samples)
  )
  
  count_matrix <- count_matrix[, rownames(colData)]
  
  dds <- DESeqDataSetFromMatrix(countData = count_matrix,
                                colData = colData,
                                design = ~ condition)
  
  keep <- rowSums(count_matrix >= 1) >= 2
  dds <- dds[keep, ]
  cat("Genes after pre-filtering:", nrow(dds), "\n")
  
  dds <- DESeq(dds)
  
  res <- results(dds,
                 contrast = c("condition", "heterograft", "selfgraft"),
                 alpha = 0.05)
  
  cat("\n--- Summary ---\n")
  print(summary(res))
  
  up <- subset(res, padj < 0.05 & log2FoldChange > 0 & !is.na(padj))
  cat("Mobile gene candidates (upregulated in heterograft):", nrow(up), "\n")
  
  dir.create(output_prefix, showWarnings = FALSE)
  write.csv(as.data.frame(res),
            paste0(output_prefix, "/DESeq2_all_results.csv"))
  write.csv(as.data.frame(up)[order(as.data.frame(up)$log2FoldChange, decreasing = TRUE), ],
            paste0(output_prefix, "/DESeq2_mobile_candidates.csv"))
  writeLines(rownames(up),
             paste0(output_prefix, "/DESeq2_mobile_geneIDs.txt"))
  
  cat("Files saved to:", output_prefix, "\n")
  return(res)
}
